#!/bin/bash

echo "üîç Verificando e instalando dependencias necesarias..."
sudo apt-get update

# Reinstala libstdc++6 por si faltan rutas
sudo apt-get install --reinstall -y libstdc++6

# Asegura que GTK est√© instalado
sudo apt-get install -y libgtk-3-dev

# Asegura que GLib est√© completo por si falla GIO
sudo apt-get install -y libglib2.0-dev

# Muestra si el archivo gtk.h existe
if [ -f /usr/include/gtk-3.0/gtk/gtk.h ]; then
    echo "‚úÖ gtk/gtk.h est√° presente."
else
    echo "‚ùå gtk/gtk.h NO se encuentra. Algo fall√≥ en la instalaci√≥n."
fi

# Crea una carpeta de prueba y compila un ejemplo
echo "üõ†Ô∏è Compilando un programa de ejemplo..."
mkdir -p ~/gtk_prueba
cat << 'EOF' > ~/gtk_prueba/hola_gtk.c
#include <gtk/gtk.h>

int main(int argc, char *argv[]) {
    gtk_init(&argc, &argv);
    GtkWidget *ventana = gtk_window_new(GTK_WINDOW_TOPLEVEL);
    gtk_window_set_title(GTK_WINDOW(ventana), "Hola GTK");
    g_signal_connect(ventana, "destroy", G_CALLBACK(gtk_main_quit), NULL);
    gtk_widget_show_all(ventana);
    gtk_main();
    return 0;
}
EOF

cd ~/gtk_prueba
gcc hola_gtk.c -o hola_gtk `pkg-config --cflags --libs gtk+-3.0`

echo "‚úÖ Listo. Ejecuta el programa con: ./hola_gtk"
